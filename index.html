Arrow Functions

▼ 
const add = (a, b) => a + b;
const square = x => x * x;
const greet = () => "Hello!";

console.log(add(3, 4));    // 7
console.log(square(5));    // 25
console.log(greet());      // Hello!

// Lexical 'this'
const timer = {
  id: 42,
  start() {
    return [1,2,3].map(n => n + this.id);
  }
};
console.log(timer.start()); // [43, 44, 45]

Destructuring

▼ 
// Object destructuring
const user = { id:1, name:"Bob", role:"admin" };
const { name, role } = user;
console.log(name, role); // Bob admin

// Array destructuring
const [r, , b] = ["red", "green", "blue"];
console.log(r, b);       // red blue

// Swap variables
let x = 1, y = 2;
[x, y] = [y, x];
console.log(x, y);       // 2 1

// Nested + defaults
const { a: { b: deep } = {}, c = "default" }
  = { a: { b: 99 } };
console.log(deep, c);    // 99 default

Spread & Rest

▼ 
// Spread — expand
const a = [1, 2, 3];
const b = [0, ...a, 4];
console.log(b); // [0,1,2,3,4]

const obj1 = { x: 1 };
const obj2 = { ...obj1, y: 2 };
console.log(obj2); // {x:1, y:2}

// Rest — collect
function sum(first, ...rest) {
  return first + rest.reduce((t,n) => t+n, 0);
}
console.log(sum(1,2,3,4)); // 10

Classes

  speak() { return `${this.name} makes a noise.`; }
}

class Dog extends Animal {
  speak() { return `${this.name} barks!`; }
}

const d = new Dog("Rex");
console.log(d.speak());
